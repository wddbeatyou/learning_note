3.opencv源码编译

```
1.构建
cmake \
  -D CMAKE_BUILD_TYPE=RELEASE \
  -D CMAKE_INSTALL_PREFIX=/usr/local \
  -D OPENCV_ENABLE_NONFREE=ON \
  -D WITH_CUDA=ON \
  -D BUILD_opencv_cudafeatures2d=ON \
  -D CUDA_ARCH_BIN="6.1 7.0 7.5 8.0 8.6 8.9" \
  -D CUDNN_LIBRARY=/home/supcon/nvidia/cudnn-linux-x86_64-8.9.7.29_cuda12-archive/lib/libcudnn.so \
  -D CUDNN_INCLUDE_DIR=/home/supcon/nvidia/cudnn-linux-x86_64-8.9.7.29_cuda12-archive/include \
  -D CUDA_TOOLKIT_ROOT_DIR=/usr/local/cuda-12.4 \
  -D CUDA_FAST_MATH=ON \
  -D WITH_CUBLAS=ON \
  -D OPENCV_DNN_CUDA=ON \
  -D OPENCV_EXTRA_MODULES_PATH=../opencv_contrib-4.10.0/modules \
  -D BUILD_opencv_python3=ON \
  -D PYTHON3_EXECUTABLE=/home/supcon/anaconda/miniconda3/envs/image_compare/bin/python3.9 \
  -D PYTHON3_INCLUDE_DIR=$(python3.9 -c "from distutils.sysconfig import get_python_inc; print(get_python_inc())") \
  -D PYTHON3_LIBRARY=/home/supcon/anaconda/miniconda3/envs/image_compare/lib/libpython3.9.so \
  -D PYTHON3_NUMPY_INCLUDE_DIRS=/home/supcon/anaconda/miniconda3/envs/cuda_py39/lib/python3.9/site-packages/numpy/core/include \
  -D BUILD_EXAMPLES=OFF \
  -D BUILD_TESTS=OFF \
  ..
2.编译
    make -j$(nproc)  # 编译（使用所有CPU核心）
    sudo make install  # 安装到系统目录
3.复制cv2.cpython-39-x86_64-linux-gnu.so文件到conda虚拟环境中
	cp /home/supcon/software/opencv-4.10.0/build/lib/python3/cv2.cpython-39-x86_64-linux-gnu.so /home/supcon/anaconda/miniconda3/envs/cuda_py39/lib/python3.9/site-packages
4.验证
    import cv2
    python -c "import cv2; print(cv2.__version__); print(cv2.cuda.getCudaEnabledDeviceCount())"
    print("OpenCV版本:", cv2.__version__)
    print("CUDA设备数量:", cv2.cuda.getCudaEnabledDeviceCount())
    print(cv2.getBuildInformation())  # 查看详细构建信息
```
